<script>
    document.addEventListener('DOMContentLoaded', function() {
        // Auto-refresh stats every 5 minutes
        setInterval(function() {
            {% if perms.users.can_manage_users %}
            fetch('/api/quick-stats/')
                .then(response => response.json())
                .then(data => {
                    // Update system stats if available
                    const statElements = document.querySelectorAll('.stat-number');
                    if (statElements.length >= 7) { // Personal + System stats
                        statElements[4].textContent = data.total_users;
                        statElements[5].textContent = data.active_users;
                        statElements[6].textContent = data.new_users_today;
                        statElements[7].textContent = data.locked_users;
                    }
                })
                .catch(error => console.log('Stats update failed:', error));
            {% endif %}
        }, 300000); // 5 minutes

        // Animate numbers on load
        const statNumbers = document.querySelectorAll('.stat-number');
        statNumbers.forEach(stat => {
            const finalNumber = parseInt(stat.textContent) || 0;
            if (finalNumber > 0) {
                animateValue(stat, 0, finalNumber, 1000);
            }
        });

        function animateValue(element, start, end, duration) {
            const range = end - start;
            const increment = range / (duration / 16);
            let current = start;

            const timer = setInterval(() => {
                current += increment;
                if (current >= end) {
                    element.textContent = end;
                    clearInterval(timer);
                } else {
                    element.textContent = Math.floor(current);
                }
            }, 16);
        }
    });
</script>
{% endblock %}

        // Real-time availability checking
        function checkAvailability(field, endpoint, resultElement, currentUserId) {
            const value = field.value.trim();
            if (value.length < 3) {
                resultElement.textContent = '';
                return;
            }

            const url = `${endpoint}?${field.name}=${encodeURIComponent(value)}&user_id=${currentUserId}`;
            fetch(url)
                .then(response => response.json())
                .then(data => {
                    resultElement.textContent = data.message;
                    resultElement.className = `availability-check ${data.available ? 'available' : 'unavailable'}`;
                })
                .catch(error => {
                    console.error('Error checking availability:', error);
                });
        }

        // Set up availability checking
        document.addEventListener('DOMContentLoaded', function() {
            const emailField = document.querySelector('#id_email');
            const usernameField = document.querySelector('#id_username');
            const emailResult = document.querySelector('#emailAvailability');
            const usernameResult = document.querySelector('#usernameAvailability');
            const currentUserId = {{ form.instance.pk }};

            if (emailField && emailResult) {
                let emailTimeout;
                emailField.addEventListener('input', function() {
                    clearTimeout(emailTimeout);
                    emailTimeout = setTimeout(() => {
                        checkAvailability(emailField, '#', emailResult, currentUserId);
                    }, 500);
                });
            }

            if (usernameField && usernameResult) {
                let usernameTimeout;
                usernameField.addEventListener('input', function() {
                    clearTimeout(usernameTimeout);
                    usernameTimeout = setTimeout(() => {
                        checkAvailability(usernameField, '#', usernameResult, currentUserId);
                    }, 500);
                });
            }

            // Track form changes url "check_username_availability"
            const form = document.getElementById('userUpdateForm');
            const originalData = new FormData(form);
            let hasChanges = false;

            form.addEventListener('input', function() {
                hasChanges = true;
            });

            form.addEventListener('change', function() {
                hasChanges = true;
            });

            // Warn before leaving if there are unsaved changes
            window.addEventListener('beforeunload', function(e) {
                if (hasChanges) {
                    e.preventDefault();
                    e.returnValue = '';
                }
            });

            // Remove warning when form is submitted
            form.addEventListener('submit', function() {
                hasChanges = false;
            });
        });

        // Auto-dismiss alerts after 5 seconds
        setTimeout(function() {
            var alerts = document.querySelectorAll('.alert');
            alerts.forEach(function(alert) {
                var bsAlert = new bootstrap.Alert(alert);
                bsAlert.close();
            });
        }, 5000);
    </script>

<script>
    document.addEventListener('DOMContentLoaded', function() {
        const checkboxes = document.querySelectorAll('.user-checkbox');
        const bulkActions = document.getElementById('bulkActions');
        const selectedCountSpan = document.getElementById('selectedCount');
        const selectedUsersInput = document.getElementById('id_selected_users');
        const selectAllBtn = document.getElementById('selectAll');

        // Add select all button
        const searchHeader = document.querySelector('.search-header h4');
        searchHeader.innerHTML += `
            <button type="button" class="btn btn-outline-light btn-sm ms-3" id="selectAllBtn">
                <i class="bi bi-check-square me-1"></i>Select All
            </button>
        `;

        const selectAllButton = document.getElementById('selectAllBtn');

        function updateBulkActions() {
            const selectedCheckboxes = document.querySelectorAll('.user-checkbox:checked');
            const count = selectedCheckboxes.length;

            selectedCountSpan.textContent = count;

            if (count > 0) {
                bulkActions.classList.add('show');
                const selectedIds = Array.from(selectedCheckboxes).map(cb => cb.value);
                selectedUsersInput.value = selectedIds.join(',');
            } else {
                bulkActions.classList.remove('show');
                selectedUsersInput.value = '';
            }

            // Update select all button
            if (count === checkboxes.length) {
                selectAllButton.innerHTML = '<i class="bi bi-square me-1"></i>Deselect All';
            } else {
                selectAllButton.innerHTML = '<i class="bi bi-check-square me-1"></i>Select All';
            }
        }

        // Handle individual checkbox changes
        checkboxes.forEach(checkbox => {
            checkbox.addEventListener('change', updateBulkActions);
        });

        // Handle select all button
        selectAllButton.addEventListener('click', function() {
            const allChecked = document.querySelectorAll('.user-checkbox:checked').length === checkboxes.length;

            checkboxes.forEach(checkbox => {
                checkbox.checked = !allChecked;
            });

            updateBulkActions();
        });

        // Auto-submit search form on select change
        const searchForm = document.getElementById('searchForm');
        const selectInputs = searchForm.querySelectorAll('select');

        selectInputs.forEach(select => {
            select.addEventListener('change', function() {
                searchForm.submit();
            });
        });

        // Real-time search with debounce
        const searchInput = searchForm.querySelector('input[name="search_query"]');
        let searchTimeout;

        if (searchInput) {
            searchInput.addEventListener('input', function() {
                clearTimeout(searchTimeout);
                searchTimeout = setTimeout(() => {
                    if (this.value.length === 0 || this.value.length >= 3) {
                        searchForm.submit();
                    }
                }, 500);
            });
        }

        // Keyboard shortcuts
        document.addEventListener('keydown', function(e) {
            // Ctrl/Cmd + A to select all
            if ((e.ctrlKey || e.metaKey) && e.key === 'a' && !e.target.matches('input, textarea')) {
                e.preventDefault();
                selectAllButton.click();
            }

            // Escape to clear selection
            if (e.key === 'Escape') {
                checkboxes.forEach(cb => cb.checked = false);
                updateBulkActions();
            }
        });

        // Confirm bulk actions
        const bulkForm = document.getElementById('bulkForm');
        bulkForm.addEventListener('submit', function(e) {
            const action = document.getElementById('id_action').value;
            const count = document.querySelectorAll('.user-checkbox:checked').length;

            let message = `Are you sure you want to ${action} ${count} selected user(s)?`;

            if (action === 'delete') {
                message = `⚠️ WARNING: This will permanently delete ${count} user(s) and cannot be undone. Are you absolutely sure?`;
            }

            if (!confirm(message)) {
                e.preventDefault();
            }
        });

        // Animate stats on load
        const statNumbers = document.querySelectorAll('.stat-number');
        statNumbers.forEach(stat => {
            const finalNumber = parseInt(stat.textContent) || 0;
            if (finalNumber > 0) {
                animateValue(stat, 0, finalNumber, 800);
            }
        });

        function animateValue(element, start, end, duration) {
            const range = end - start;
            const increment = range / (duration / 16);
            let current = start;

            const timer = setInterval(() => {
                current += increment;
                if (current >= end) {
                    element.textContent = end;
                    clearInterval(timer);
                } else {
                    element.textContent = Math.floor(current);
                }
            }, 16);
        }
    });

    function hideBulkActions() {
        document.getElementById('bulkActions').classList.remove('show');
        document.querySelectorAll('.user-checkbox').forEach(cb => cb.checked = false);
        document.getElementById('selectedCount').textContent = '0';
        document.getElementById('id_selected_users').value = '';
    }

    // Export functionality
    function exportUsers() {
        const selectedCheckboxes = document.querySelectorAll('.user-checkbox:checked');

        if (selectedCheckboxes.length > 0) {
            const selectedIds = Array.from(selectedCheckboxes).map(cb => cb.value);
            const form = document.createElement('form');
            form.method = 'POST';
            form.action = '{% url "export_users" %}';

            const csrfInput = document.createElement('input');
            csrfInput.type = 'hidden';
            csrfInput.name = 'csrfmiddlewaretoken';
            csrfInput.value = getCookie('csrftoken');
            form.appendChild(csrfInput);

            const idsInput = document.createElement('input');
            idsInput.type = 'hidden';
            idsInput.name = 'user_ids';
            idsInput.value = selectedIds.join(',');
            form.appendChild(idsInput);

            document.body.appendChild(form);
            form.submit();
            document.body.removeChild(form);
        } else {
            window.location.href = '{% url "export_users" %}';
        }
    }
</script>

    <script>
        // Real-time availability checking
        function checkAvailability(field, endpoint, resultElement) {
            const value = field.value.trim();
            if (value.length < 3) {
                resultElement.textContent = '';
                return;
            }

            fetch(`${endpoint}?${field.name}=${encodeURIComponent(value)}`)
                .then(response => response.json())
                .then(data => {
                    resultElement.textContent = data.message;
                    resultElement.className = `availability-check ${data.available ? 'available' : 'unavailable'}`;
                })
                .catch(error => {
                    console.error('Error checking availability:', error);
                });
        }

        // Set up availability checking
        document.addEventListener('DOMContentLoaded', function() {
            const emailField = document.querySelector('#id_email');
            const usernameField = document.querySelector('#id_username');
            const emailResult = document.querySelector('#emailAvailability');
            const usernameResult = document.querySelector('#usernameAvailability');

            if (emailField && emailResult) {
                let emailTimeout;
                emailField.addEventListener('input', function() {
                    clearTimeout(emailTimeout);
                    emailTimeout = setTimeout(() => {
                        checkAvailability(emailField, '#', emailResult);
                    }, 500);
                });
            }

            if (usernameField && usernameResult) {
                let usernameTimeout;
                usernameField.addEventListener('input', function() {
                    clearTimeout(usernameTimeout);
                    usernameTimeout = setTimeout(() => {
                        checkAvailability(usernameField, '#', usernameResult);
                    }, 500);
                });
            }

            // Form validation // "check_username_availability" %}
            const form = document.getElementById('userCreateForm');
            form.addEventListener('submit', function(e) {
                const requiredFields = form.querySelectorAll('[required]');
                let isValid = true;

                requiredFields.forEach(field => {
                    if (!field.value.trim()) {
                        isValid = false;
                        field.classList.add('is-invalid');
                    } else {
                        field.classList.remove('is-invalid');
                    }
                });

                if (!isValid) {
                    e.preventDefault();
                    alert('Please fill in all required fields.');
                }
            });
        });

        // Auto-dismiss alerts after 5 seconds
        setTimeout(function() {
            var alerts = document.querySelectorAll('.alert');
            alerts.forEach(function(alert) {
                var bsAlert = new bootstrap.Alert(alert);
                bsAlert.close();
            });
        }, 5000);
    </script>

    <script>
        class SignatureManager {
            constructor() {
                this.canvas = document.getElementById('signatureCanvas');
                this.ctx = this.canvas.getContext('2d');
                this.isDrawing = false;
                this.lastX = 0;
                this.lastY = 0;

                this.initializeCanvas();
                this.bindEvents();
            }

            initializeCanvas() {
                this.ctx.strokeStyle = '#000000';
                this.ctx.lineWidth = 2;
                this.ctx.lineCap = 'round';
                this.ctx.lineJoin = 'round';

                // Set canvas background to white
                this.ctx.fillStyle = 'white';
                this.ctx.fillRect(0, 0, this.canvas.width, this.canvas.height);
            }

            bindEvents() {
                // Canvas drawing events
                this.canvas.addEventListener('mousedown', (e) => this.startDrawing(e));
                this.canvas.addEventListener('mousemove', (e) => this.draw(e));
                this.canvas.addEventListener('mouseup', () => this.stopDrawing());
                this.canvas.addEventListener('mouseout', () => this.stopDrawing());

                // Touch events for mobile
                this.canvas.addEventListener('touchstart', (e) => {
                    e.preventDefault();
                    const touch = e.touches[0];
                    const mouseEvent = new MouseEvent('mousedown', {
                        clientX: touch.clientX,
                        clientY: touch.clientY
                    });
                    this.canvas.dispatchEvent(mouseEvent);
                });

                this.canvas.addEventListener('touchmove', (e) => {
                    e.preventDefault();
                    const touch = e.touches[0];
                    const mouseEvent = new MouseEvent('mousemove', {
                        clientX: touch.clientX,
                        clientY: touch.clientY
                    });
                    this.canvas.dispatchEvent(mouseEvent);
                });

                this.canvas.addEventListener('touchend', (e) => {
                    e.preventDefault();
                    const mouseEvent = new MouseEvent('mouseup', {});
                    this.canvas.dispatchEvent(mouseEvent);
                });

                // Controls
                document.getElementById('penColor').addEventListener('change', (e) => {
                    this.ctx.strokeStyle = e.target.value;
                });

                document.getElementById('penSize').addEventListener('input', (e) => {
                    this.ctx.lineWidth = e.target.value;
                });

                document.getElementById('clearCanvas').addEventListener('click', () => {
                    this.clearCanvas();
                });

                // Typed signature
                document.getElementById('typedSignature').addEventListener('input', this.updateTypedPreview);
                document.getElementById('fontStyle').addEventListener('change', this.updateTypedPreview);

                // File upload
                const fileInput = document.querySelector('input[type="file"]');
                const uploadArea = document.getElementById('uploadArea');

                uploadArea.addEventListener('click', () => fileInput.click());
                uploadArea.addEventListener('dragover', this.handleDragOver);
                uploadArea.addEventListener('drop', this.handleDrop);

                // Preview and save
                document.getElementById('previewSignature').addEventListener('click', this.previewSignature);
                document.getElementById('confirmSave').addEventListener('click', this.saveSignature);
            }

            getMousePos(e) {
                const rect = this.canvas.getBoundingClientRect();
                return {
                    x: (e.clientX - rect.left) * (this.canvas.width / rect.width),
                    y: (e.clientY - rect.top) * (this.canvas.height / rect.height)
                };
            }

            startDrawing(e) {
                this.isDrawing = true;
                const pos = this.getMousePos(e);
                this.lastX = pos.x;
                this.lastY = pos.y;
                this.canvas.classList.add('active');
            }

            draw(e) {
                if (!this.isDrawing) return;

                const pos = this.getMousePos(e);

                this.ctx.beginPath();
                this.ctx.moveTo(this.lastX, this.lastY);
                this.ctx.lineTo(pos.x, pos.y);
                this.ctx.stroke();

                this.lastX = pos.x;
                this.lastY = pos.y;
            }

            stopDrawing() {
                this.isDrawing = false;
                this.canvas.classList.remove('active');
            }

            clearCanvas() {
                this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
                this.ctx.fillStyle = 'white';
                this.ctx.fillRect(0, 0, this.canvas.width, this.canvas.height);
            }

            updateTypedPreview() {
                const text = document.getElementById('typedSignature').value;
                const font = document.getElementById('fontStyle').value;
                const preview = document.getElementById('typedPreview');

                if (text) {
                    preview.innerHTML = `<span style="font-family: ${font}; font-size: 3rem; color: #000;">${text}</span>`;
                } else {
                    preview.innerHTML = '<span class="text-muted">Your typed signature will appear here</span>';
                }
            }

            handleDragOver(e) {
                e.preventDefault();
                e.currentTarget.classList.add('dragover');
            }

            handleDrop(e) {
                e.preventDefault();
                e.currentTarget.classList.remove('dragover');
                const files = e.dataTransfer.files;
                if (files.length > 0) {
                    document.querySelector('input[type="file"]').files = files;
                }
            }

            previewSignature() {
                const activeTab = document.querySelector('.nav-link.active').id;
                const previewContent = document.getElementById('previewContent');
                let hasSignature = false;

                if (activeTab === 'draw-tab') {
                    const canvasData = this.canvas.toDataURL();
                    if (this.isCanvasBlank()) {
                        alert('Please draw a signature first.');
                        return;
                    }
                    previewContent.innerHTML = `<img src="${canvasData}" style="max-width: 100%; height: auto;">`;
                    hasSignature = true;
                } else if (activeTab === 'upload-tab') {
                    const fileInput = document.querySelector('input[type="file"]');
                    if (fileInput.files.length === 0) {
                        alert('Please select an image file first.');
                        return;
                    }
                    const file = fileInput.files[0];
                    const reader = new FileReader();
                    reader.onload = function(e) {
                        previewContent.innerHTML = `<img src="${e.target.result}" style="max-width: 100%; height: auto;">`;
                    };
                    reader.readAsDataURL(file);
                    hasSignature = true;
                } else if (activeTab === 'type-tab') {
                    const text = document.getElementById('typedSignature').value;
                    if (!text.trim()) {
                        alert('Please type your signature first.');
                        return;
                    }
                    const font = document.getElementById('fontStyle').value;
                    previewContent.innerHTML = `<span style="font-family: ${font}; font-size: 3rem; color: #000;">${text}</span>`;
                    hasSignature = true;
                }

                if (hasSignature) {
                    new bootstrap.Modal(document.getElementById('previewModal')).show();
                }
            }

            isCanvasBlank() {
                const blank = document.createElement('canvas');
                blank.width = this.canvas.width;
                blank.height = this.canvas.height;
                const blankCtx = blank.getContext('2d');
                blankCtx.fillStyle = 'white';
                blankCtx.fillRect(0, 0, blank.width, blank.height);

                return this.canvas.toDataURL() === blank.toDataURL();
            }

            saveSignature() {
                const activeTab = document.querySelector('.nav-link.active').id;

                if (activeTab === 'draw-tab') {
                    if (this.isCanvasBlank()) {
                        alert('Please draw a signature first.');
                        return;
                    }
                    document.getElementById('signatureData').value = this.canvas.toDataURL();
                } else if (activeTab === 'type-tab') {
                    const text = document.getElementById('typedSignature').value;
                    if (!text.trim()) {
                        alert('Please type your signature first.');
                        return;
                    }
                    // Create a canvas for typed signature
                    const tempCanvas = document.createElement('canvas');
                    tempCanvas.width = 600;
                    tempCanvas.height = 200;
                    const tempCtx = tempCanvas.getContext('2d');
                    tempCtx.fillStyle = 'white';
                    tempCtx.fillRect(0, 0, tempCanvas.width, tempCanvas.height);
                    tempCtx.fillStyle = 'black';
                    tempCtx.font = `48px ${document.getElementById('fontStyle').value}`;
                    tempCtx.textAlign = 'center';
                    tempCtx.fillText(text, tempCanvas.width/2, tempCanvas.height/2 + 16);
                    document.getElementById('signatureData').value = tempCanvas.toDataURL();
                }

                // Close modal and submit form
                bootstrap.Modal.getInstance(document.getElementById('previewModal')).hide();
                document.getElementById('signatureForm').submit();
            }
        }

        // Initialize signature manager when page loads
        document.addEventListener('DOMContentLoaded', function() {
            new SignatureManager();

            // Auto-dismiss alerts after 5 seconds
            setTimeout(function() {
                var alerts = document.querySelectorAll('.alert');
                alerts.forEach(function(alert) {
                    var bsAlert = new bootstrap.Alert(alert);
                    bsAlert.close();
                });
            }, 5000);
        });
    </script>
{% endblock %}

<script>
    document.addEventListener('DOMContentLoaded', function() {
        const avatarUpload = document.getElementById('avatarUpload');
        const avatarInput = document.getElementById('{{ form.avatar.id_for_label }}');
        const uploadPreview = document.getElementById('uploadPreview');
        const profileForm = document.getElementById('profileForm');

        // File upload preview
        avatarInput.addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (file) {
                if (file.size > 5 * 1024 * 1024) { // 5MB limit
                    alert('File size must be less than 5MB');
                    return;
                }

                if (!file.type.startsWith('image/')) {
                    alert('Please select an image file');
                    return;
                }

                const reader = new FileReader();
                reader.onload = function(e) {
                    uploadPreview.src = e.target.result;
                    uploadPreview.style.display = 'block';
                };
                reader.readAsDataURL(file);
            }
        });

        // Drag and drop functionality
        avatarUpload.addEventListener('dragover', function(e) {
            e.preventDefault();
            this.classList.add('dragover');
        });

        avatarUpload.addEventListener('dragleave', function(e) {
            e.preventDefault();
            this.classList.remove('dragover');
        });

        avatarUpload.addEventListener('drop', function(e) {
            e.preventDefault();
            this.classList.remove('dragover');

            const files = e.dataTransfer.files;
            if (files.length > 0) {
                avatarInput.files = files;
                avatarInput.dispatchEvent(new Event('change'));
            }
        });

        // Form validation
        const requiredFields = profileForm.querySelectorAll('[required]');
        const submitBtn = profileForm.querySelector('button[type="submit"]');

        function validateForm() {
            let isValid = true;

            requiredFields.forEach(field => {
                if (!field.value.trim()) {
                    isValid = false;
                    field.classList.add('is-invalid');
                } else {
                    field.classList.remove('is-invalid');
                }
            });

            // Phone number validation
            const phoneField = document.getElementById('{{ form.phone_number.id_for_label }}');
            if (phoneField && phoneField.value) {
                if (!validatePhone(phoneField.value)) {
                    isValid = false;
                    phoneField.classList.add('is-invalid');
                } else {
                    phoneField.classList.remove('is-invalid');
                }
            }

            submitBtn.disabled = !isValid;
            return isValid;
        }

        // Real-time validation
        requiredFields.forEach(field => {
            field.addEventListener('input', validateForm);
            field.addEventListener('blur', validateForm);
        });

        // Auto-save functionality (optional)
        let autoSaveTimeout;
        const formInputs = profileForm.querySelectorAll('input, textarea, select');

        formInputs.forEach(input => {
            input.addEventListener('input', function() {
                clearTimeout(autoSaveTimeout);
                autoSaveTimeout = setTimeout(() => {
                    // Show auto-save indicator
                    showAutoSaveIndicator();
                }, 2000);
            });
        });

        function showAutoSaveIndicator() {
            const indicator = document.createElement('div');
            indicator.className = 'alert alert-info alert-dismissible fade show position-fixed';
            indicator.style.cssText = 'top: 20px; right: 20px; z-index: 9999; max-width: 300px;';
            indicator.innerHTML = `
                <i class="bi bi-cloud-check me-2"></i>
                Auto-save available
                <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
            `;
            document.body.appendChild(indicator);

            setTimeout(() => {
                if (indicator.parentNode) {
                    indicator.remove();
                }
            }, 5000);
        }

        // Form submission with loading state
        profileForm.addEventListener('submit', function(e) {
            if (!validateForm()) {
                e.preventDefault();
                return;
            }

            submitBtn.innerHTML = '<i class="spinner-border spinner-border-sm me-2"></i>Saving...';
            submitBtn.disabled = true;

            // Re-enable button after 5 seconds in case of error
            setTimeout(() => {
                submitBtn.innerHTML = '<i class="bi bi-check-circle me-2"></i>Save Changes';
                submitBtn.disabled = false;
            }, 5000);
        });

        // Character counter for bio
        const bioField = document.getElementById('{{ form.bio.id_for_label }}');
        if (bioField) {
            const maxLength = 500;
            const counter = document.createElement('div');
            counter.className = 'text-muted small mt-1 text-end';
            bioField.parentNode.appendChild(counter);

            function updateCounter() {
                const remaining = maxLength - bioField.value.length;
                counter.textContent = `${bioField.value.length}/${maxLength} characters`;
                counter.className = `text-${remaining < 50 ? 'danger' : 'muted'} small mt-1 text-end`;
            }

            bioField.addEventListener('input', updateCounter);
            updateCounter();
        }

        // Timezone detection
        const timezoneField = document.getElementById('{{ form.timezone.id_for_label }}');
        if (timezoneField && !timezoneField.value) {
            const detectedTimezone = Intl.DateTimeFormat().resolvedOptions().timeZone;
            if (detectedTimezone) {
                timezoneField.value = detectedTimezone;

                // Show notification
                const notification = document.createElement('div');
                notification.className = 'alert alert-info alert-dismissible fade show';
                notification.innerHTML = `
                    <i class="bi bi-geo-alt me-2"></i>
                    We've detected your timezone as <strong>${detectedTimezone}</strong>
                    <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
                `;
                timezoneField.parentNode.insertBefore(notification, timezoneField);
            }
        }

        // Keyboard shortcuts
        document.addEventListener('keydown', function(e) {
            // Ctrl/Cmd + S to save
            if ((e.ctrlKey || e.metaKey) && e.key === 's') {
                e.preventDefault();
                profileForm.requestSubmit();
            }

            // Ctrl/Cmd + R to reset
            if ((e.ctrlKey || e.metaKey) && e.key === 'r') {
                e.preventDefault();
                if (confirm('Are you sure you want to reset the form?')) {
                    profileForm.reset();
                    uploadPreview.style.display = 'none';
                }
            }
        });

        // Initial validation
        validateForm();
    });
</script>

    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const sidebar = document.getElementById('sidebar');
            const sidebarToggle = document.getElementById('sidebarToggle');
            const loadingOverlay = document.querySelector('.loading');

            // Sidebar toggle functionality
            if (sidebarToggle && sidebar) {
                sidebarToggle.addEventListener('click', function() {
                    sidebar.classList.toggle('show');
                });

                // Close sidebar when clicking outside on mobile
                document.addEventListener('click', function(event) {
                    if (window.innerWidth <= 768) {
                        if (!sidebar.contains(event.target) && !sidebarToggle.contains(event.target)) {
                            sidebar.classList.remove('show');
                        }
                    }
                });
            }

            // Auto-hide loading spinner
            if (loadingOverlay) {
                setTimeout(() => {
                    loadingOverlay.style.display = 'none';
                }, 500);
            }

            // Animate stat numbers on load
            const statNumbers = document.querySelectorAll('.stat-number');
            statNumbers.forEach(stat => {
                const finalNumber = parseInt(stat.textContent) || 0;
                if (finalNumber > 0 && finalNumber < 10000) {
                    animateValue(stat, 0, finalNumber, 1500);
                }
            });

            // Add fade-in animation to elements
            const fadeElements = document.querySelectorAll('.fade-in');
            fadeElements.forEach((element, index) => {
                element.style.animationDelay = `${index * 0.1}s`;
            });

            // Enhanced dropdown functionality
            const dropdownToggle = document.querySelector('.dropdown-toggle');
            if (dropdownToggle) {
                dropdownToggle.addEventListener('click', function(e) {
                    e.stopPropagation();
                });
            }

            // Add smooth scrolling to anchor links
            document.querySelectorAll('a[href^="#"]').forEach(anchor => {
                anchor.addEventListener('click', function(e) {
                    e.preventDefault();
                    const target = document.querySelector(this.getAttribute('href'));
                    if (target) {
                        target.scrollIntoView({
                            behavior: 'smooth',
                            block: 'start'
                        });
                    }
                });
            });

            // Responsive sidebar handling
            function handleResize() {
                if (window.innerWidth > 768) {
                    sidebar.classList.remove('show');
                }
            }

            window.addEventListener('resize', handleResize);

            // Add loading states to action buttons
            const actionButtons = document.querySelectorAll('.action-btn');
            actionButtons.forEach(button => {
                button.addEventListener('click', function(e) {
                    // Add subtle loading effect
                    this.style.opacity = '0.7';
                    this.style.transform = 'scale(0.98)';

                    setTimeout(() => {
                        this.style.opacity = '';
                        this.style.transform = '';
                    }, 200);
                });
            });

            // Enhanced security status animations
            const securityItems = document.querySelectorAll('.security-item');
            securityItems.forEach((item, index) => {
                item.style.animationDelay = `${0.5 + index * 0.1}s`;
                item.classList.add('fade-in');
            });

            // Activity items animation
            const activityItems = document.querySelectorAll('.activity-item');
            activityItems.forEach((item, index) => {
                item.style.animationDelay = `${0.3 + index * 0.15}s`;
                item.classList.add('fade-in');
            });

            // Stat cards hover effects enhancement
            const statCards = document.querySelectorAll('.stat-card');
            statCards.forEach(card => {
                card.addEventListener('mouseenter', function() {
                    this.style.transform = 'translateY(-8px) scale(1.02)';
                });

                card.addEventListener('mouseleave', function() {
                    this.style.transform = 'translateY(0) scale(1)';
                });
            });

            // Auto-refresh stats every 5 minutes (if needed)
            setInterval(function() {
                // Uncomment and modify as needed for your API
                /*
                fetch('/api/dashboard-stats/')
                    .then(response => response.json())
                    .then(data => {
                        // Update stat numbers
                        updateStatNumbers(data);
                    })
                    .catch(error => console.log('Stats update failed:', error));
                */
            }, 300000); // 5 minutes

            // Function to animate numbers
            function animateValue(element, start, end, duration) {
                const range = end - start;
                const increment = range / (duration / 16);
                let current = start;

                const timer = setInterval(() => {
                    current += increment;
                    if (current >= end) {
                        element.textContent = end;
                        clearInterval(timer);
                    } else {
                        element.textContent = Math.floor(current);
                    }
                }, 16);
            }

            // Function to update stat numbers (for auto-refresh)
            function updateStatNumbers(data) {
                const statElements = document.querySelectorAll('.stat-number');
                if (data.total_companies !== undefined) statElements[0].textContent = data.total_companies;
                if (data.total_employees !== undefined) statElements[1].textContent = data.total_employees;
                if (data.total_branches !== undefined) statElements[2].textContent = data.total_branches;
                if (data.total_domains !== undefined) statElements[3].textContent = data.total_domains;
            }

            // Enhanced keyboard navigation
            document.addEventListener('keydown', function(e) {
                // Toggle sidebar with Ctrl/Cmd + B
                if ((e.ctrlKey || e.metaKey) && e.key === 'b') {
                    e.preventDefault();
                    sidebar.classList.toggle('show');
                }

                // Close sidebar with Escape
                if (e.key === 'Escape' && sidebar.classList.contains('show')) {
                    sidebar.classList.remove('show');
                }
            });

            const statusBadges = document.querySelectorAll('.status-badge');

                if (typeof bootstrap !== 'undefined' && bootstrap.Tooltip) {
                    new bootstrap.Tooltip(badge);
                }

                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.classList.add('fade-in');
                        observer.unobserve(entry.target);
                    }
                });
            });

                observer.observe(el);
            });

    <script src="https://cdnjs.cloudflare.com/ajax/libs/bootstrap/5.3.0/js/bootstrap.bundle.min.js"></script>
    <script>
        class TwoFactorSetup {
            constructor() {
                this.currentStep = 1;
                this.maxStep = 4;
                this.bindEvents();
            }

            bindEvents() {
                // Verification code input formatting
                const verificationInput = document.querySelector('#id_verification_code');
                if (verificationInput) {
                    verificationInput.addEventListener('input', this.formatVerificationCode);
                    verificationInput.addEventListener('keypress', this.handleKeyPress);
                }

                // Backup confirmation checkbox
                document.getElementById('confirmBackup').addEventListener('change', (e) => {
                    document.getElementById('enableBtn').disabled = !e.target.checked;
                });

                // Form submission
                document.getElementById('twoFactorForm').addEventListener('submit', this.handleSubmit);
            }

            formatVerificationCode(e) {
                let value = e.target.value.replace(/\D/g, ''); // Remove non-digits
                if (value.length > 6) value = value.slice(0, 6); // Limit to 6 digits
                e.target.value = value;
            }

            handleKeyPress(e) {
                // Only allow digits
                if (!/[0-9]/.test(e.key) && !['Backspace', 'Delete', 'Tab', 'Escape', 'Enter'].includes(e.key)) {
                    e.preventDefault();
                }
            }

            nextStep(step) {
                if (step > this.maxStep) return;

                // Hide current step
                document.getElementById(`step${this.currentStep}-content`).classList.remove('active');
                document.getElementById(`step${this.currentStep}-indicator`).classList.remove('active');
                document.getElementById(`step${this.currentStep}-indicator`).classList.add('completed');

                // Show next step
                this.currentStep = step;
                document.getElementById(`step${this.currentStep}-content`).classList.add('active');
                document.getElementById(`step${this.currentStep}-indicator`).classList.remove('inactive');
                document.getElementById(`step${this.currentStep}-indicator`).classList.add('active');

                // Focus on verification input if on step 3
                if (step === 3) {
                    setTimeout(() => {
                        const input = document.querySelector('#id_verification_code');
                        if (input) input.focus();
                    }, 300);
                }

                // Generate backup codes if on step 4
                if (step === 4) {
                    this.generateBackupCodes();
                }
            }

            prevStep(step) {
                if (step < 1) return;

                // Hide current step
                document.getElementById(`step${this.currentStep}-content`).classList.remove('active');
                document.getElementById(`step${this.currentStep}-indicator`).classList.remove('active');
                document.getElementById(`step${this.currentStep}-indicator`).classList.add('inactive');

                // Show previous step
                this.currentStep = step;
                document.getElementById(`step${this.currentStep}-content`).classList.add('active');
                document.getElementById(`step${this.currentStep}-indicator`).classList.remove('completed');
                document.getElementById(`step${this.currentStep}-indicator`).classList.add('active');
            }

            verifyCode() {
                const codeInput = document.querySelector('#id_verification_code');
                const code = codeInput.value;

                if (code.length !== 6) {
                    this.showError('Please enter a 6-digit verification code.');
                    codeInput.focus();
                    return;
                }

                // Show loading state
                const verifyBtn = document.querySelector('button[onclick="verifyCode()"]');
                const originalContent = verifyBtn.innerHTML;
                verifyBtn.innerHTML = '<i class="fas fa-spinner fa-spin me-2"></i>Verifying...';
                verifyBtn.disabled = true;

                // Simulate verification (in real implementation, this would be an AJAX call)
                setTimeout(() => {
                    // For demo purposes, accept any 6-digit code
                    if (code === '123456' || code.length === 6) {
                        this.showSuccess('Code verified successfully!');
                        setTimeout(() => {
                            this.nextStep(4);
                        }, 1000);
                    } else {
                        this.showError('Invalid verification code. Please try again.');
                        verifyBtn.innerHTML = originalContent;
                        verifyBtn.disabled = false;
                        codeInput.focus();
                    }
                }, 1500);
            }

            generateBackupCodes() {
                const codes = [];
                for (let i = 0; i < 6; i++) {
                    const code = this.generateRandomCode();
                    codes.push(code);
                }

                const container = document.getElementById('backupCodes');
                container.innerHTML = '';

                codes.forEach(code => {
                    const col = document.createElement('div');
                    col.className = 'col-md-6';
                    col.innerHTML = `<div class="backup-code">${code}</div>`;
                    container.appendChild(col);
                });
            }

            generateRandomCode() {
                const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
                let result = '';
                for (let i = 0; i < 12; i++) {
                    if (i > 0 && i % 4 === 0) result += '-';
                    result += chars.charAt(Math.floor(Math.random() * chars.length));
                }
                return result;
            }

            downloadBackupCodes() {
                const codes = Array.from(document.querySelectorAll('.backup-code')).map(el => el.textContent);
                const content = `Two-Factor Authentication Backup Codes\n\nIMPORTANT: Keep these codes secure and accessible.\nEach code can only be used once.\n\n${codes.join('\n')}\n\nGenerated: ${new Date().toLocaleString()}`;

                const blob = new Blob([content], { type: 'text/plain' });
                const url = window.URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = 'backup-codes.txt';
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                window.URL.revokeObjectURL(url);
            }

            showSuccess(message) {
                this.showAlert(message, 'success');
            }

            showError(message) {
                this.showAlert(message, 'danger');
            }

            showAlert(message, type) {
                const alertContainer = document.querySelector('.position-fixed');
                if (!alertContainer) return;

                const alert = document.createElement('div');
                alert.className = `alert alert-${type} alert-dismissible fade show`;
                alert.innerHTML = `
                    ${message}
                    <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
                `;

                alertContainer.appendChild(alert);

                // Auto-dismiss after 5 seconds
                setTimeout(() => {
                    if (alert.parentNode) {
                        const bsAlert = new bootstrap.Alert(alert);
                        bsAlert.close();
                    }
                }, 5000);
            }

            handleSubmit(e) {
                const form = e.target;
                const submitBtn = form.querySelector('button[type="submit"]');

                // Show loading state
                submitBtn.innerHTML = '<i class="fas fa-spinner fa-spin me-2"></i>Enabling 2FA...';
                submitBtn.disabled = true;

                // Add success animation to the card
                setTimeout(() => {
                    document.querySelector('.setup-card').classList.add('success-animation');
                }, 500);

                return true;
            }
        }

        // Global functions for onclick handlers
        function nextStep(step) {
            window.twoFactorSetup.nextStep(step);
        }

        function prevStep(step) {
            window.twoFactorSetup.prevStep(step);
        }

        function verifyCode() {
            window.twoFactorSetup.verifyCode();
        }

        function downloadBackupCodes() {
            window.twoFactorSetup.downloadBackupCodes();
        }

        function copyToClipboard(text) {
            navigator.clipboard.writeText(text).then(() => {
                window.twoFactorSetup.showSuccess('Code copied to clipboard!');
            }).catch(() => {
                // Fallback for older browsers
                const textArea = document.createElement('textarea');
                textArea.value = text;
                document.body.appendChild(textArea);
                textArea.select();
                document.execCommand('copy');
                document.body.removeChild(textArea);
                window.twoFactorSetup.showSuccess('Code copied to clipboard!');
            });
        }

        // Initialize when page loads
        document.addEventListener('DOMContentLoaded', function() {
            window.twoFactorSetup = new TwoFactorSetup();

            // Auto-dismiss existing alerts after 5 seconds
            setTimeout(function() {
                var alerts = document.querySelectorAll('.alert');
                alerts.forEach(function(alert) {
                    var bsAlert = new bootstrap.Alert(alert);
                    bsAlert.close();
                });
            }, 5000);
        });
    </script>

    <script>
        // Plan distribution chart
        const planData = {{ plan_stats|safe }};
        const ctx = document.getElementById('planChart').getContext('2d');

        let currentChart = null;

        function createChart(type) {
            if (currentChart) {
                currentChart.destroy();
            }

            const labels = planData.map(item => item.plan__display_name || item.plan__name || 'Unknown');
            const data = planData.map(item => item.count);
            const colors = [
                '#007bff', '#28a745', '#ffc107', '#dc3545',
                '#6f42c1', '#fd7e14', '#20c997', '#6c757d'
            ];

            const config = {
                type: type,
                data: {
                    labels: labels,
                    datasets: [{
                        label: 'Companies',
                        data: data,
                        backgroundColor: colors.slice(0, data.length),
                        borderColor: colors.slice(0, data.length),
                        borderWidth: 2
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: {
                            position: type === 'pie' ? 'right' : 'top',
                        },
                        tooltip: {
                            callbacks: {
                                label: function(context) {
                                    const total = context.dataset.data.reduce((a, b) => a + b, 0);
                                    const percentage = ((context.parsed * 100) / total).toFixed(1);
                                    return `${context.label}: ${context.parsed} (${percentage}%)`;
                                }
                            }
                        }
                    },
                    scales: type === 'bar' ? {
                        y: {
                            beginAtZero: true,
                            ticks: {
                                stepSize: 1
                            }
                        }
                    } : {}
                }
            };

            currentChart = new Chart(ctx, config);
        }

        // Initialize with pie chart
        createChart('pie');

        // Handle chart type change
        document.querySelectorAll('input[name="chartType"]').forEach(radio => {
            radio.addEventListener('change', function() {
                const chartType = this.id === 'chartPie' ? 'pie' : 'bar';
                createChart(chartType);
            });
        });

        // Auto-refresh data every 5 minutes
        setInterval(function() {
            location.reload();
        }, 300000);
    </script>
